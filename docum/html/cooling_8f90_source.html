75 !> @brief Loads radiative cooling coefficients from a data file
76 !> @details Loads radiative cooling coefficients from a data file into the
77 !! cooldata global array. The number of data points must defined in the first
78 !! line of the data file, and every subsequent line must have a pair of data
79 !! values containing the logarithms of temperature and the cooling rate, with
80 !! the latter multiplied by -1 so it's positive.
81 !! Example table:
82 !! 5
83 !! 4.0 23.4

85 !! 5.0 22.5

99

101  open (unit=99, file=cooling_file, status='old', iostat=istat)

104  write(logu,*) "***ABORTING***"

220 !> @brief Applies radiative cooling to a block
221 !> @details This employs the previously loaded tabulated cooling function.
222 !! Cooling is turned off below for temperatures below the minimum value
223 !! of the table (global var cool_Tmin), while above the maximum value
224 !! (cool_Tmax) free-free emission, propto sqrt(T), is used.
225 !! Cooling is applied by decreasing the pressure of the cell by the factor
226 !! exp(-n^2*Lambda(T)*dt/e0)
227 !! where lambda(T) is the temperature-dependent cooling function (in units of
228 !! erg cm^3 s^-1), n is the number density, dt is the timestep and e0 is the
229 !! initial thermal energy density of the cell. This comes from taking the
230 !! energy density loss rate, -n^2*Lambda(T), and assuming Lambda(T) is approx.
231 !! linear throughout the small interval dt.
232 !! The cooling factor is limited to being at least as large as the parameter
233 !! cooling_limit. Also, a locally-defined temperature floor is enforced.
234 !> @param bIndx The local index of the block
235 !> @param maxloss Maximum fractional thermal energy loss observed
236 subroutine apply_cooling (bIndx, maxloss)
237 
238  use parameters
239  use globals
240  implicit none
241 
242  integer, intent(in) :: bindx
243  real, intent(out) :: maxloss
244 
245  real, parameter :: t_floor = 10.0
246 
247  real :: temp, new_temp, logt, vel2, eth, ek, aloss, numdens, ce, cool_factor
248  real :: frac_loss, metal
249  integer :: i, j, k
250 
251  maxloss = 0.0
252 
253  do i=1,ncells_x
254  do j=1,ncells_y
255  do k=1,ncells_z
256 
257  ! Calculate temperature of this cell
258  call calctemp(prim(bindx,:,i,j,k), temp)
259  logt = log10(temp)
260 
261  ! Obtain metallicity of this cell (for metal cooling)
262  if (cooling_type.eq.cool_table_metal) then
263  metal = abs(prim(bindx,metalpas,i,j,k)/prim(bindx,1,i,j,k))
264  end if
265 
266  ! Cooling not applied below cool_Tmin
267  if (logt.ge.cool_tmin) then
268 
269  ! Code units
270  eth = cv * prim(bindx,5,i,j,k)
271  vel2 = prim(bindx,2,i,j,k)**2 + &
272  prim(bindx,3,i,j,k)**2 + &
273  prim(bindx,4,i,j,k)**2
274  ek = 0.5 * prim(bindx,1,i,j,k) * vel2
275 
276  ! Interpolate cooling coefficient from table
277  if (cooling_type.eq.cool_table) then
278  call find_coef(logt, aloss)
279  else if (cooling_type.eq.cool_table_metal) then
280  call find_coef_metal(logt, metal, aloss)
281  end if
282 
283  ! Calculate radiative loss and new temperature
284  numdens = prim(bindx,1,i,j,k)*d_sc/(mui*amu) ! cgs, gas ionized
285  ce = aloss*(numdens**2)/(eth*e_sc) ! cgs
286  cool_factor = exp(-ce*(dt*t_sc))
287  frac_loss = 1.0-cool_factor
288 
289  ! DEBUG
290 ! write(logu,*) localBlocks(bIndx), i, j, k
291 ! write(logu,*) log10(temp), aloss
292 ! write(logu,*) numdens, frac_loss
293 
294  ! Record maximum cooling for this block before limiting
295  maxloss = max(maxloss, frac_loss)
296 
297  ! Limit cool_factor directly, if needed
298  if (cool_factor.lt.1.0-cooling_limit) then
299  cool_factor = 1.0-cooling_limit
300  end if
301 
302  ! Impose a temperature floor by adjusting cool_factor, if needed
303  ! Set to 10 K by default
304  new_temp = temp * cool_factor
305  if (new_temp.lt.t_floor) then
306  new_temp = t_floor
307  cool_factor = t_floor / temp
308  end if
309 
310  ! Update pressure and total energy
311  prim(bindx,5,i,j,k) = prim(bindx,5,i,j,k) * cool_factor
312  eth = cv * prim(bindx,5,i,j,k)
313  u(bindx,5,i,j,k) = ek + eth
314 
315  end if
316 
317  end do
318  end do
319  end do
320 
321 end subroutine apply_cooling

325 !> @brief Interpolates a cooling cofficient from 1D temperature-tabulated data
326 !> @param logT The base-10 logarithm of the temperature (Kelvin)
327 !> @param coef Interpolated cooling coefficient
328 subroutine find_coef (logT, coef)
329 
330  use globals, only: cooltable, nptst, cool_tmin, cool_tmax
331  implicit none
332 
333  real, intent(in) :: logt
334  real, intent(out) :: coef
335 
336  integer :: i
337  real :: t0, t1, c0, c1
338 
339  ! Below Tmin, the returned coefficient is zero (no cooling)
340  if (logt.lt.cool_tmin) then
341 
342  coef = 0.0
343  return
344 
345  ! Above Tmax, we assume the coefficient is proportional to sqrt(T)
346  ! (corresponding to the free-free regime)
347  else if (logt.ge.cool_tmax) then
348 
349 ! coef = cooltable(2,nptsT)*sqrt(real(T)/cool_Tmax)
350  coef = 10**( cooltable(2,nptst) + 0.5*(logt-cool_tmax) )
351  return
352 
353  ! For T in range of the table, we do linear interpolation
354  else
355 
356  do i=2,nptst
357  if (cooltable(1,i).gt.logt) then
358  t0 = cooltable(1,i-1)
359  c0 = cooltable(2,i-1)
360  t1 = cooltable(1,i)
361  c1 = cooltable(2,i)
362  coef = 10**( (c1-c0)/(t1-t0)*(logt-t0) + c0 )
363  return
364  end if
365  end do
366 
367  end if
368 
369  end subroutine find_coef

373 !> @brief Interpolates a cooling cofficient from 2D temperature-metallicity
374 !! tabulated data
375 !> @param logT Base-10 logarithm of temperature (Kelvin)
376 !> @param Z Metallicity (fraction of metals)
377 !> @param coef Interpolated cooling coefficient
378 !> @details The subroutines assumes cooltable holds 2D tabulated coefficients
379 !! in the form cooltable(T,Z), where each dimension holds nptsT+1 and nptsZ+1
380 !! values. The first row and the first column state the temperature/metallicity
381 !! values, while the inner rows and columns hold the corresponding coefficients.
382 !! Below Tmin, cooling is set to zero. Above Tmax, the linear interpolation
383 !! with respect to temperature is changed to an extrapolation using a ~sqrt(T)
384 !! law. Extrapolation over metallicity is linear on both ends.
385 subroutine find_coef_metal (logT, Z, coef)
386 
387  use globals, only: cooltable, nptst, nptsz, cool_tmin, cool_tmax, cool_zmin, cool_zmax
388  implicit none
389 
390  real, intent(in) :: logt
391  real, intent(in) :: z
392  real, intent(out) :: coef
393 
394  integer :: tlo, thi, tmid, zlo, zhi, zmid
395  real :: t0, t1, z0, z1, c0, c1, tc, zc
396 
397  ! Below Tmin, the returned coefficient is zero (no cooling)
398  if (logt.lt.cool_tmin) then
399  coef = 0.0
400  return
401  end if
402 
403  ! Find the T and Z brackets in which the given values are located
404  if (logt.le.cool_tmax) then
405  ! Binary search
406  tlo = 2
407  thi = nptst+1
408  do while ((thi-tlo).gt.1)
409  tmid = (tlo+thi)/2
410  if (logt <= cooltable(tmid,1)) then
411  thi = tmid
412  else
413  tlo = tmid
414  end if
415  end do
416  t0 = cooltable(tlo,1)
417  t1 = cooltable(thi,1)
418  end if
419 
420  if (z.lt.cool_zmin) then
421  zlo = 2
422  zhi = 3
423  z0 = cooltable(1,zlo)
424  z1 = cooltable(1,zhi)
425  else if (z.gt.cool_zmax) then
426  zlo = nptsz
427  zhi = nptsz+1
428  z0 = cooltable(1,zlo)
429  z1 = cooltable(1,zhi)
430  else
431  ! Binary search
432  zlo = 2
433  zhi = nptsz+1
434  do while ((zhi-zlo).gt.1)
435  zmid = (zlo+zhi)/2
436  if (z <= cooltable(1,zmid)) then
437  zhi = zmid
438  else
439  zlo = zmid
440  end if
441  end do
442  z0 = cooltable(1,zlo)
443  z1 = cooltable(1,zhi)
444  end if
445 
446  ! Above Tmax, we assume the coefficient is proportional to sqrt(T),
447  ! but still linearly interpolate/extrapolate over Z
448  if (logt.ge.cool_tmax) then
449 
450  c0 = cooltable(nptst,zlo) + 0.5*(logt-cool_tmax)
451  c1 = cooltable(nptst,zhi) + 0.5*(logt-cool_tmax)
452  coef = 10**( (c1-c0)/(z1-z0)*(z-z0) + c0 )
453  return
454 
455  ! For T in range, we do bilinear interpolation over T and Z
456  else
457 
458  tc = (logt-t0)/(t1-t0)
459  zc = (z-z0)/(z1-z0)
460  coef = (1.0-tc)*(1.0-zc)*cooltable(tlo,zlo) + &
461  (1.0-tc)*zc*cooltable(tlo,zhi) + &
462  tc*(1.0-zc)*cooltable(thi,zlo) + &
463  tc*zc*cooltable(thi,zhi)
464  coef = 10**coef
465  return
466 
467  end if
468 
469  end subroutine find_coef_metal
