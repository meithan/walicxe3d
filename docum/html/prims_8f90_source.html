234 !! primitive variables. Passive scalars are simply copied over.
235 !> @param pvars(neqtot) An input vector containing (rho,u,v,w,P,s1,s2,...)
236 !> @param uvars(neqtot) An output vector containing (rho,rho*u,rho*v,rho*w,E,s1,s2,...)
237 subroutine prim2flow (pvars, uvars)
238

243  real, intent(out) :: uvars(neqtot)
244 
245  real :: v2
246 
247  uvars(1) = pvars(1)
248  uvars(2) = pvars(1)*pvars(2)

272 !> @brief Calculates the vector of primitive variables given the vector of
273 !! conserved (flow) variables. Any passive scalars are simply copied over.
274 !> @param uvars(neqtot) An input vector containing (rho,rho*u,rho*v,rho*w,E,...)
275 !> @param pvars(neqtot) An output vector containing (rho,u,v,w,P,...)
276 !> @param istat An output integer indicating the success status of the
277 !! operation (0=all good, 1=pressure floored, 2=density floored).
278 subroutine flow2prim (uvars, pvars, istat)
279 
280  use parameters
281  use globals ! DEBUG
282  implicit none
283 
284  real, intent(in) :: uvars(neqtot)
285  real, intent(out) :: pvars(neqtot)
286  integer, intent(out) :: istat
287 
288  real :: rhov2
289 
290  istat = 0
291 
292  if (uvars(1).eq.0.0) then
293  write(logu,*) "Received zero density!!"
294  call clean_abort(error_division_by_zero)
295  end if
296 
297  pvars(1) = uvars(1)
298  pvars(2) = uvars(2)/uvars(1)
299  pvars(3) = uvars(3)/uvars(1)
300  pvars(4) = uvars(4)/uvars(1)
301 
302  rhov2 = (uvars(2)**2 + uvars(3)**2 + uvars(4)**2)/uvars(1)
303  pvars(5) = (uvars(5)-0.5*rhov2)/cv
304 
305  ! Floor on pressure
306  if (pvars(5).lt.1.0e-30) then
307  pvars(5) = 1.0e-30
308  istat = 1
309  end if
310 
311  ! Floor on density
312  if (pvars(1).lt.1.0e-40) then
313  pvars(1) = 1.0e-40
314  istat = 2
315  end if
316 
317 #ifdef PASBP
318  pvars(6) = uvars(6)
319  pvars(7) = uvars(7)
320  pvars(8) = uvars(8)
321 #endif
322 
323  if (npassive.ge.1) then
324  pvars(firstpas:neqtot) = uvars(firstpas:neqtot)
325  end if
326 
327  return
328 
329 end subroutine flow2prim

333 !> @brief Calculates the vectors of physical fluxes along a specific dimension
334 !! given the vector of primitive variables.
335 !> @param pvars(neqtot) An input vector containing primitives
336 !> @param dimf Dimension along which fluxes are to be calculated
337 !> @param F(neqtot) An output vector containing fluxes along dimf
338 subroutine prim2fluxes (pvars, dimens, flux)
339 
340  use parameters
341  implicit none
342 
343  real, intent(in) :: pvars(neqtot)
344  integer, intent(in) :: dimens
345  real, intent(out) :: flux(neqtot)
346 
347  real :: v2, etot
348  real :: pvars1(neqtot)
349 
350  ! Make a copy of the primitives so the original ones are not split
351  pvars1(:) = pvars(:)
352 
353  ! Swap velocity components when dimf=2 (Y) or dimf=3(Z)
354  if (dimens.eq.dim_y) then
355  call swapxy(pvars1)
356  else if (dimens.eq.dim_z) then
357  call swapxz(pvars1)
358  end if
359 
360  v2 = pvars1(2)**2 + pvars1(3)**2 + pvars1(4)**2
361  etot = 0.5*pvars1(1)*v2 + cv*pvars1(5)
362 
363  ! Calculate Fluxes (formulae for X)
364  flux(1) = pvars1(1)*pvars1(2)
365  flux(2) = pvars1(5) + pvars1(1)*(pvars1(2)**2)
366  flux(3) = pvars1(1)*pvars1(2)*pvars1(3)
367  flux(4) = pvars1(1)*pvars1(2)*pvars1(4)
368  flux(5) = pvars1(2)*(etot+pvars1(5))
369 #ifdef PASBP
370  flux(6) = 0
371  flux(7) = pvars1(2)*pvars1(7) - pvars1(3)*pvars1(6)
372  flux(8) = pvars1(2)*pvars1(8) - pvars1(4)*pvars1(6)
373 #endif
374 
375  if (npassive.ge.1) then
376  flux(firstpas:neqtot) = pvars1(2)*pvars1(firstpas:neqtot)
377  end if
378 
379  ! Swap flux components to get correct fluxes for Y or Z
380  if (dimens.eq.dim_y) then
381  call swapxy(flux)
382  else if (dimens.eq.dim_z) then
383  call swapxz(flux)
384  end if
385 
386  return
387 
388 end subroutine prim2fluxes
