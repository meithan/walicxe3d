28 !! variables and subroutines required to impose a spherical wind source.

30 !! How to use this module.

32 !! 1) The parameters of each wind must be defined by creating an object of

115  !> @brief Imposes a spherical wind source on the simulation
116  !> @details Simulates a spherical wind source by setting the flow
117  !! conditions in a region of given radius centered at (xc,yc,zc),
118  !! in which a steady-state wind solution is imposed with the given
119  !! mass-loss rate, terminal speed and temperature.
120  !> @param wind_params A spherical_wind_type variable containing the
121  !! wind parameters. See the module documentation for further details.

133  integer :: nb, bid, i, j, k

135  real :: dens, vx, vy, vz, pres, x, y, z, dist, mu, metal
136  real :: primit(neqtot)
137  real :: zone(6)
138  integer :: zlevel
139 
140  write(logu,*) ""
141  write(logu,'(1x,a)') "> Imposing spherical wind source ..."
142 
143  ! Unpack wind source parameters
144  xc = wind_params%xc
145  yc = wind_params%yc
146  zc = wind_params%zc
147  vwx = wind_params%vx

155  metal = wind_params%metal

157  ! Report wind parameters
158  ! TODO: REPORT ONLY ON FIRST CALL

160  write(logu,'(1x,a,es12.5)') "vinf = ", vinf
161  write(logu,'(1x,a,es12.5)') "Radius = ", radius
162  write(logu,'(1x,a,es12.5)') "Rho(R) = ", mdot/vinf/radius/radius/(4*pi)
163  write(logu,'(1x,a,es12.5)') "Temp = ", temp
164  write(logu,'(1x,a,es12.5,1x,es12.5,1x,es12.5)') "Location: ", xc, yc, zc
165 
166  ! Refine the zone around the wind source to provide adequate resolution
167  if (it.eq.0) then
168  write(logu,*) "(winds)it=",it
169  zone(1) = xc - radius
170  zone(2) = xc + radius
171  zone(3) = yc - radius
172  zone(4) = yc + radius
173  zone(5) = zc - radius
174  zone(6) = zc + radius
175  zlevel = maxlev
176  call refinezone(zone, zlevel)
177  end if
178 
179  ! Impose flow conditions, where applicable
180  do nb=1,nbmaxproc
181  bid = localblocks(nb)
182  if (bid.ne.-1) then
183 
184  do i=nxmin,nxmax
185  do j=nymin,nymax
186  do k=nzmin,nzmax
187 
188  call cellpos(bid, i, j, k, x, y, z)
189  x = x*l_sc; y = y*l_sc; z = z*l_sc
190  dist = sqrt((x-xc)**2+(y-yc)**2+(z-zc)**2) ! phys units
191 
192  if (dist.lt.radius) then
193 
194  ! Calculate wind density, velocity and pressure in this cell
195  dens = mdot/vinf/dist/dist/(4.0*pi)
196  vx = vinf*(x-xc)/dist + vwx
197  vy = vinf*(y-yc)/dist + vwy
198  vz = vinf*(z-zc)/dist + vwz
199  pres= dens/(mu*amu)*kb*temp
200 
201  ! DEBUG
202 ! write(logu,*) dens, vx/1e5, vy/1e5, vz/1e5, pres
203  ! DEBUG
204 
205  ! Scale primitives
206  primit(1) = dens/d_sc
207  primit(2) = vx/v_sc
208  primit(3) = vy/v_sc
209  primit(4) = vz/v_sc
210  primit(5) = pres/p_sc
211 #ifdef PASBP
212  ! Magnetic field
213  primit(6) = 0.0
214  primit(7) = 0.0
215  primit(8) = 0.0
216 #endif
217  ! Passive scalar for metalicity
218  if (cooling_type.eq.cool_table_metal) then
219  primit(metalpas) = metal*primit(1)
220  end if
221 
222  ! Convert primitives and set flow vars for this cell
223  call prim2flow( primit, uvars(nb,:,i,j,k) )
224 
225  end if
226 
227  end do
228  end do
229  end do
230 
231  end if
232  end do
