26 !> @brief Wrapper for Godunov-type numerical solvers
27 !> @details Computes intercell numerical fluxes using any Riemann solver and
28 !! advances the flow variables one timestep. The solver can be set to be
29 !! first or second order (in space and time). This is a high-level wrapper.
30 subroutine godunov (order)
31

43  ! 1st-order timestep
44 
45  ! Halve timestep for first part of second-order HLL
46  if (order.eq.2) then
47  dtp = dt/2.0
48  else
49  dtp = dt
50  end if
51 
52  ! Exchange 1-deep boundaries to fill ghost cells (all blocks)
53  call tic(mark)
54  write(logu,*) ""
55  write(logu,'(1x,a)') "> Exchanging 1-deep boundary layers ..."
56  call boundary(1, u)
57  write(logu,*) "Boundaries exchanged in", nicetoc(mark)

63  bcount = 0
64  call tic(mark)
65  write(logu,*) ""

67  write(logu,'(1x,a)') "> Integrating blocks ..."
68  else if (order.eq.2) then

74  if (bid.ne.-1) then

77  select case (solver_type)
78

81

84 
85  case (solver_hllc)
86  call hllcfluxes(bindx, 1)
87 
88  end select

92 
93  bcount = bcount + 1

95  end if
96  end do
97

99 
100  ! -----------------------------------
101  ! 2nd-order full timestep (skipped in 1st-order schemes)
102 
103  if (order.eq.2) then
104 
105  ! Exchange 2-deep boundaries to fill ghost cells (all blocks)
106  call tic(mark)
107  write(logu,*) ""
108  write(logu,'(1x,a)') "> Exchanging 2-deep boundary layers ..."
109  call boundary(2, up)
110  write(logu,*) "Boundaries exchanged in", nicetoc(mark)
111 
112  ! Update primitives in ghost cells
113  call calcprimsall(up, prim, cells_ghost)

156 !> @brief Applies conservative formula to calculate stepped U variables
157 !> @details This is the standard upwing Godunov step. Assumes the numerical
158 !! intercell fluxes FC have been calculated for this block
159 !> @param locIndx Block's local index
160 !> @param dtp Timestep to use
161 subroutine upwindstep (locIndx, dtp)
162 
163  use parameters
164  use globals
165  implicit none
166 
167  integer, intent(in) :: locindx
168  real, intent(in) :: dtp
169 
170  integer :: i, j, k, bid, lev
171  integer :: ieq ! DEBUG
172  real :: dtdx, dtdy, dtdz
173 
174  bid = localblocks(locindx)
175  call meshlevel(bid, lev)
176 
177  dtdx = dtp/dx(lev)
178  dtdy = dtp/dy(lev)
179  dtdz = dtp/dz(lev)
180 
181  ! Apply conservative formula (10.2 of Toro) to obtain UPs
182  do i=1,ncells_x
183  do j=1,ncells_y
184  do k=1,ncells_z
185 ! DEBUG
186 ! Check FC, GC and HC values for NaNs
187 do ieq=1,neqtot
188 if (fc(ieq,i-1,j,k).ne.fc(ieq,i-1,j,k)) then
189  write(logu,'(a,i2,i2,i2)') "Nan in FC at ", i-1, j, k
190 end if
191 if (fc(ieq,i,j,k).ne.fc(ieq,i,j,k)) then
192  write(logu,'(a,i2,i2,i2)') "Nan in FC at ", i, j, k
193 end if
194 if (gc(ieq,i,j-1,k).ne.gc(ieq,i,j-1,k)) then
195  write(logu,'(a,i2,i2,i2)') "Nan in GC at ", i, j-1, k
196 end if
197 if (gc(ieq,i,j,k).ne.gc(ieq,i,j,k)) then
198  write(logu,'(a,i2,i2,i2)') "Nan in GC at ", i, j, k
199 end if
200 if (hc(ieq,i,j,k-1).ne.hc(ieq,i,j,k-1)) then
201  write(logu,'(a,i2,i2,i2)') "Nan in HC at ", i, j, k-1
202 end if
203 if (hc(ieq,i,j,k).ne.hc(ieq,i,j,k)) then
204  write(logu,'(a,i2,i2,i2)') "Nan in HC at ", i, j, k
205 end if
206 end do
207 ! DEBUG
208  up(locindx,:,i,j,k) &
209  = u(locindx,:,i,j,k) &
210  + dtdx*(fc(:,i-1,j,k)-fc(:,i,j,k)) &
211  + dtdy*(gc(:,i,j-1,k)-gc(:,i,j,k)) &
212  + dtdz*(hc(:,i,j,k-1)-hc(:,i,j,k))
213  end do
214  end do
215  end do
216 
217 end subroutine upwindstep

221 !> @brief Obtains wavespeed estimates given primitives at interface
222 !> @details Computes wavespeed estimates for the two outer shock waves and the
223 !! contact discontinuity (if needed) in the Riemann fan. This routine employs
224 !! the x-component of speed.
225 !> @param dl Density at "left" of interface
226 !> @param dr Density at "right" of interface
227 !> @param ul Velocity at "left" of interface
228 !> @param ur Velocity at "right" of interface
229 !> @param pl Pressure at "left" of interface
230 !> @param pr Pressure at "right" of interface
231 !> @param sl Wavespeed estimate for "left" moving wave
232 !> @param sr Wavespeed estimate for "right" moving wave
233 !> @param sst Wavespeed estimate for contact discontinuity
234 subroutine wavespeed (primL, primR, sl, sr, sst)
235 
236  use parameters
237  implicit none
238 
239  real, intent(in) :: priml(neqtot)
240  real, intent(in) :: primr(neqtot)
241  real, intent(out) :: sl
242  real, intent(out) :: sr
243  real, intent(out) :: sst
244 
245  real :: dl1, ul1, pl1, cl
246  real :: dr1, ur1, pr1, cr
247 
248  ! Unpack primitives
249  dl1 = priml(1)
250  dr1 = primr(1)
251  ul1 = priml(2)
252  ur1 = primr(2)
253  pl1 = priml(5)
254  pr1 = primr(5)
255 
256  ! Sound speeds
257  call sound(priml,cl)
258  call sound(primr,cr)
259 
260  ! Compute SL and SR
261  ! Davis direct bounded, 10.38 of Toro
262  sl = min( ul1-cl, ur1-cr )
263  sr = max( ul1+cl, ur1+cr )
264 
265  ! Compute S* (HLLC only)
266  if (solver_type.eq.solver_hllc) then
267  sst = (pr1-pl1 + dl1*ul1*(sl-ul1) - dr1*ur1*(sr-ur1)) / &
268  (dl1*(sl-ul1)-dr1*(sr-ur1))
269  else
270  sst = 0.0
271  end if
272 
273  return
274 
275 end subroutine wavespeed

314  real function average (a,b,opt)
315 
316  use constants
317  implicit none
318 
319  real, intent(in) :: a, b
320  integer, intent(in) :: opt
321 
322  real :: s, c, d, eps
323 
324  select case (opt)
325 
326  case (limiter_none)
327  average = 0.5*(a+b)
328 
329  case (limiter_vanleer)
330  if (a*b.le.0.0) then
331  average = 0.0
332  else
333  average = a*b*(a+b)/(a*a+b*b)
334  end if
335 
336  case (limiter_minmod)
337  s = sign(1.0,a)
338  average = s*max(0.0, min(abs(a), s*b))
339 
340  case (limiter_albada) ! NOT WORKING
341  eps = 1.0e-7
342  average = (a*(b*b+eps)+b*(a*a+eps))/(a*a+b*b*eps)
343 
344  case (limiter_umist)
345  s = sign(1.0,a)
346  c = 0.25*a + 0.75*b
347  d = 0.75*a + 0.25*b
348  average = min(2.0*abs(a), 2.0*s*b, s*c, s*d)
349  average = s*max(0.0, average)
350 
351  case (limiter_woodward)
352  s = sign(1.0,a)
353  c = 0.5*(a+b)
354  average = min(2.0*abs(a), 2*s*b, s*c)
355  average = s*max(0.0, average)
356 
357  case (limiter_superbee)
358  s = sign(1.0,b)
359  c = min(2.0*abs(b), s*a)
360  d = min(abs(b),2.0*s*a)
361  average = s*max(0.0,c,d)
362 
363  case default
364  average = 0.0
365  write(*,'(a)') "WARNING: no averaging in limiter!"
366  write(*,'(a,i2)') "Passed limiter value: ", opt
367 
368  end select
369 
370  end function average
371 
372 end subroutine limiter
